#include "unity.h"
#include "Bytecode.h"
#include "rCALL.h"
#include "CException.h"

void setUp() {}
void tearDown() {}

void test_rcall_should_pop_the_next_value() {

  Instruction inst = {
                      .mnemonic = RCALL,
                      .name = "rcall"
                     };	
  Bytecode code = { .instruction = &inst,
                    .operand1 =	-1,  //address of file select register
                    .operand2 =	-1, 	  
                    .operand3 = -1,	
					.absoluteAddress = 0x100					
                  };


	rcall(&code);
	
	TEST_ASSERT_EQUAL(9,FSR[STKPTR]);
	
}
